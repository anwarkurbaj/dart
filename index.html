// Future — يرجع قيمة بعد وقت Future<String> fetchName() async { await Future.delayed(Duration(seconds: 2)); return 'أحمد'; } // استخدام async/await void main() async { print('قبل'); String name = await fetchName(); // انتظر النتيجة print('الاسم: $name'); print('بعد'); } // try/catch مع async Future<void> loadData() async { try { final data = await fetchFromApi(); print(data); } catch (e) { print('خطأ: $e'); } finally { print('انتهى'); // دايماً يشتغل } }

class Person { // ✅ Properties (الخصائص) String name; int age; String? email; // nullable // ✅ Private property (بـ _) String _password = ''; // ✅ Constructor Person({required this.name, required this.age, this.email}); // ✅ Named Constructor Person.guest() : name = 'ضيف', age = 0; // ✅ Getter String get info => '$name ($age سنة)'; // ✅ Setter set password(String val) { if (val.length >= 6) _password = val; } // ✅ Method void greet() { print('مرحباً، أنا $name'); } // ✅ toString override @override String toString() => 'Person(name: $name, age: $age)'; } // الاستخدام void main() { var person = Person(name: 'أحمد', age: 25); person.greet(); // مرحباً، أنا أحمد print(person.info); // أحمد (25 سنة) person.password = 'abc123'; // Setter var guest = Person.guest(); // Named Constructor }

📋 Abstract Class

كلاس لا يمكن إنشاء object منه مباشرة — بس يعرّف الـ interface:

abstract class Shape { // دالة مجردة — لازم كل ابن يطبّقها double area(); double perimeter(); // دالة عادية — اختيارية للـ override void describe() => print('مساحتي: ${area()}'); } class Circle extends Shape { final double radius; Circle(this.radius); @override double area() => 3.14 * radius * radius; @override double perimeter() => 2 * 3.14 * radius; } class Rectangle extends Shape { final double width, height; Rectangle(this.width, this.height); @override double area() => width * height; @override double perimeter() => 2 * (width + height); }
🔌 Interface — implements
// كل كلاس في Dart ممكن يشتغل كـ interface abstract class Printable { void print(); } abstract class Saveable { void save(); } // implements يرث الـ interface (لازم تطبق كل الدوال) class Document implements Printable, Saveable { @override void print() => print('طباعة المستند'); @override void save() => print('حفظ المستند'); } // الفرق بين extends و implements // extends → ترث الكود + الـ interface // implements → ترث الـ interface فقط (لازم تعيد الكتابة)
🧩 Mixin — إضافة وظائف بدون وراثة
mixin Swimming { void swim() => print('يسبح'); } mixin Flying { void fly() => print('يطير'); } class Duck extends Animal with Swimming, Flying { Duck() : super('بطة'); } void main() { var duck = Duck(); duck.swim(); // من Swimming duck.fly(); // من Flying }
